 60     function create_housing_layer(result) {
 61 
 62         var geojsonMarkerOptions = {
 63             radius: 8,
 64             fillColor: "grey",
 65             color: "#000",
 66             weight: 1,
 67             opacity: 1,
 68             fillOpacity: 1,
 69         };
 70 
 71         function onEachFeature(feature, layer) {
 72             if (feature.properties && feature.properties.property_type){
 73                 layer.bindPopup("Type of affordable housing unit: " + feature.properties.property_type);
 74             }
 75         };
 76 
 77         var affordable_housing_layer = L.geoJSON(result,
 78         {
 79             pointToLayer: function (feature, latlng) {
 80                     return L.circleMarker(latlng, geojsonMarkerOptions);
 81                 },
 82             pane: "pane450_2",
 83             onEachFeature: onEachFeature
 84         }
 85         ).addTo(map);
 86 
 87         overlayMaps.affordable_housing = affordable_housing_layer;
 88     };

123     /** create neighborhood layer */
124 
125     function create_neighborhood_layer(result) {
126 
127         /** neighborhood layer style */
128 
129         function zoomToFeature(e) {
130             map.fitBounds(e.target.getBounds());
131         }
132 
133         function highlightFeature(e) {
134                 var layer = e.target;
135 
136                 layer.setStyle({
137                     weight: 5,
138                     color: '#666',
139                     dashArray: '',
140                     fillOpacity: 0.7
141                 });
142 
143                 if (!L.Browser.ie && !L.Browser.opera && !L.Browser.edge) {
144                     layer.bringToFront();
145                 }
146             };
147 
148         function resetHighlight(e) {
149             console.log("fired");
150             neighborhoods_layer.resetStyle(e.target);
151         };
152 
153         function style(feature) {
154             return {
155                 fillColor: '#027192',
156                 weight: 2,
157                 opacity: 1,
158                 color: 'white',
159                 dashArray: '3',
160                 fillOpacity: 0.5,
161             };
162         };
163 
164         function onEachFeature(feature, layer) {
165             if (feature.properties && feature.properties.pri_neigh){
166                 layer.bindPopup(feature.properties.pri_neigh);
167             }
168             
169             layer.on({
170                 mouseover: highlightFeature,
171                 mouseout: resetHighlight,
172                 click: zoomToFeature
173             });
174         };
175 
176         var neighborhoods_layer = L.geoJSON(result,
177             {   
178                 pane: "pane250",
179                 style: style,
180                 onEachFeature: onEachFeature
181             }).addTo(map);
182 
183         overlayMaps.neighborhoods = neighborhoods_layer;
184 
185         create_map_layers();
186     };
